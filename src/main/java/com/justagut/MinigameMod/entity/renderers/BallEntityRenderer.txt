package com.justagut.MinigameMod.entity.renderers;

import com.justagut.MinigameMod.entity.client.TennisBallModel;
import com.justagut.MinigameMod.entity.custom.BallEntity;
import com.justagut.MinigameMod.entity.custom.TennisBall;
import com.justagut.MinigameMod.minigamemod;
import com.mojang.blaze3d.vertex.PoseStack;
import com.mojang.blaze3d.vertex.VertexConsumer;
import net.minecraft.client.renderer.MultiBufferSource;
import net.minecraft.client.renderer.RenderType;
import net.minecraft.client.renderer.entity.EntityRenderer;
import net.minecraft.client.renderer.entity.EntityRendererProvider;
import net.minecraft.client.renderer.texture.OverlayTexture;
import net.minecraft.resources.ResourceLocation;
import net.minecraft.world.entity.Entity;

public class BallEntityRenderer extends EntityRenderer<BallEntity> {

    private final BallEntityRenderer model;

    public BallEntityRenderer(EntityRendererProvider.Context context) {
        super(context);
        this.model = new TennisBall(context.bakeLayer(TennisBallModel.LAYER_LOCATION));
        this.shadowRadius = 2f;
    }

    @Override
    public ResourceLocation getTextureLocation(BallEntity entity) {
        return ResourceLocation.fromNamespaceAndPath(minigamemod.MODID,"textures/yourpath/yourtexture.png");
    }


    @Override
    public void render(BallEntity entity, float entityYaw, float partialTick, PoseStack poseStack, MultiBufferSource bufferSource, int packedLight) {

        poseStack.pushPose();

        VertexConsumer vertexConsumer = bufferSource.getBuffer(RenderType.entityCutoutNoCull(this.getTextureLocation(entity)));

        TennisBallModel.renderToBuffer(poseStack, vertexConsumer, packedLight,
                OverlayTexture.NO_OVERLAY);

        poseStack.popPose();
        super.render(entity, entityYaw, partialTick, poseStack, bufferSource, packedLight);
    }
}
